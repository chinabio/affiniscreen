# Runtime dependencies for the AffiniScreen GUI + analysis.
#
# NOTE: Do NOT pip-install parmed / pytraj here -- Amber provides them via
# PYTHONPATH, and pip-installing can shadow/break the Amber build. On an HPC
# node, prefer the conda environment (environment.yml).
#
# Amber-22 ships parmed/pytraj built against NumPy 1.x, so NumPy is pinned <2.
numpy<2
scipy
pandas<3
matplotlib

# GUI
streamlit
nglview

# Site config (optional: JSON fallback works without it)
PyYAML

# Free-energy analysis
MDAnalysis
pymbar
alchemlyb
